    materialized='table',
    tags=['mk_api']
) }}

WITH base AS (

    SELECT
        acct.acct_id_pk,
        acct.acct_id,
        acct.record_source,
        sp.sp_id,
        prem.prem_id,
        kv.kvit,
        cl.client_id,

        first_value(gp.global_pao)
            OVER (
                PARTITION BY acct.acct_id_pk
                ORDER BY gp.dbt_valid_from DESC
            ) AS global_pao,

        first_value(sactd.client_type_desc)
            OVER (
                PARTITION BY acct.acct_id_pk
                ORDER BY sactd.dbt_valid_from DESC
            ) AS client_type

    FROM {{ ref('h_acct') }} acct

    JOIN {{ ref('l_mainsa_acct') }} msa
        ON msa.acct_id_pk = acct.acct_id_pk

    JOIN {{ ref('h_sa') }} sa
        ON sa.sa_id_pk = msa.main_sa_id_pk

    JOIN (
        SELECT DISTINCT
            s_h_sa_status_flag.sa_id_pk,
            first_value(s_h_sa_status_flag.sa_status_flag)
                OVER (
                    PARTITION BY s_h_sa_status_flag.sa_id_pk
                    ORDER BY s_h_sa_status_flag.effdt DESC
                ) AS sa_status_flag
        FROM {{ ref('s_h_sa_status_flag') }} s_h_sa_status_flag
        WHERE s_h_sa_status_flag.dbt_valid_to IS NULL
    ) status
        ON status.sa_id_pk = sa.sa_id_pk
        AND status.sa_status_flag = ANY (ARRAY['20'::bpchar, '30'::bpchar, '50'::bpchar])

    JOIN (
        SELECT DISTINCT
            l_h_sp_sa.sa_id_pk,
            first_value(l_h_sp_sa.sp_id_pk)
                OVER (
                    PARTITION BY l_h_sp_sa.sa_id_pk
                    ORDER BY l_h_sp_sa.effdt DESC
                ) AS sp_id_pk
        FROM {{ ref('l_h_sp_sa') }} l_h_sp_sa
        WHERE l_h_sp_sa.dbt_valid_to IS NULL
    ) sp_sa
        ON sp_sa.sa_id_pk = sa.sa_id_pk

    JOIN {{ ref('h_sp') }} sp
        ON sp.sp_id_pk = sp_sa.sp_id_pk

    JOIN (
        SELECT DISTINCT
            l_h_sp_prem.sp_id_pk,
            first_value(l_h_sp_prem.prem_id_pk)
                OVER (
                    PARTITION BY l_h_sp_prem.sp_id_pk
                    ORDER BY l_h_sp_prem.effdt DESC NULLS LAST
                ) AS prem_id_pk
        FROM {{ ref('l_h_sp_prem') }} l_h_sp_prem
    ) sp_prem
        ON sp.sp_id_pk = sp_prem.sp_id_pk

    JOIN {{ ref('h_prem') }} prem
        ON prem.prem_id_pk = sp_prem.prem_id_pk

    LEFT JOIN {{ ref('l_acct_kvit') }} kv_link
        ON kv_link.acct_id_pk = acct.acct_id_pk

    LEFT JOIN {{ ref('h_kvit') }} kv
        ON kv.kvit_pk = kv_link.kvit_pk

    LEFT JOIN (
        SELECT DISTINCT
            l_h_acct_client.acct_id_pk,
            first_value(l_h_acct_client.client_id_pk)
                OVER (
                    PARTITION BY l_h_acct_client.acct_id_pk
                    ORDER BY l_h_acct_client.effdt DESC
                ) AS client_id_pk
        FROM {{ ref('l_h_acct_client') }} l_h_acct_client
        WHERE l_h_acct_client.dbt_valid_to IS NULL
    ) acct_cl
        ON acct_cl.acct_id_pk = acct.acct_id_pk

    LEFT JOIN {{ ref('h_client') }} cl
        ON cl.client_id_pk = acct_cl.client_id_pk

    LEFT JOIN {{ ref('s_acct_global_pao') }} gp
        ON gp.acct_id_pk = acct.acct_id_pk
        AND gp.dbt_valid_to IS NULL

    LEFT JOIN {{ ref('s_acct_client_type_descr') }} sactd
        ON sactd.acct_id_pk = acct.acct_id_pk
        AND sactd.dbt_valid_to IS NULL
)

SELECT
    md5(
        coalesce(base.sp_id::text, '') || '-' ||
        coalesce(base.prem_id::text, '')
    ) AS pk_id,

    base.sp_id AS service_point_id,
    base.prem_id AS premise_id,
    lower(base.record_source) AS system_source,

    CASE
        WHEN base.client_type = 'Физические лица'::text THEN 'ФЛ'::text
        WHEN base.client_type = 'Юридические лица'::text THEN 'ЮЛ'::text
        ELSE 'ЮЛ'::text
    END AS client_type,

    base.global_pao,

    string_agg(DISTINCT base.acct_id::text, ','::text) AS contract_id_list,
    string_agg(DISTINCT base.kvit, ','::text) AS kvit_list,
    string_agg(DISTINCT base.client_id::text, ','::text) AS organization_id_list,

    string_agg(
        DISTINCT ((base.global_pao || '|'::text) || base.kvit),
        ','::text
    ) AS accounts_id_with_pao

FROM base

GROUP BY
    base.sp_id,
    base.prem_id,
    lower(base.record_source),
    base.client_type,
    base.global_pao
